Cでの実装例(nC2のとき)

for (int i = 0; i < n*(n-1)/2; i++) {
    int j = n-1, s = 0;
    while (s + j <= i) s += j--;
    printf("%d, %d\n", n-1-j, n-j+i-s);
}

出力(n=4)

0, 1
0, 2
0, 3
1, 2
1, 3
